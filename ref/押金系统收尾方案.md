```typescript

// 2. 押金记录列表
listDeposits(
  userId: string,
  page: number,     // 默认 1
  limit: number,    // 默认 20

// 3. 押金余额查询
getBalance(
  userId: string,
): Promise<{ balance: number; refundingCount: number; tipList: string[] }>

// 4. 押金退款申请
refundDeposit(
  userId: string,
  amount: number,
  alipayNo?: string,
  alipayRealname?: string,
): Promise<{ orderNo: string }>

// 5. 处理押金支付成功回调（供支付模块调用，不对外暴露）
handleDepositPaymentSuccess(orderNo: string, paymentId: string): Promise<void>

// 6. 扣减押金（供竞标模块调用，不对外暴露）
deductDeposit(userId: string, amount: number, remark: string): Promise<void>
```

### 对外暴露的 API（4 个）

`handleDepositPaymentSuccess` 和 `deductDeposit` 供内部模块调用，不暴露 REST API。

## 2. 控制器设计

遵循项目现有风格（参照 `PaymentsController`）：
- 路由前缀：`api/v1/deposit`
- 使用 `JwtAuthGuard` 鉴权
- 使用 `@nestjs/swagger` 装饰器
- 响应格式：`{ success: true, data: ... }`
- 请求参数通过 `@Req() req: AuthenticatedRequest` 获取 userId

### 2.1 创建押金充值订单

```
POST /api/v1/deposit/create
```

**鉴权：** JwtAuthGuard @ApiBearerAuth()

**请求体：**

```typescript
// src/deposit/dto/create-deposit.dto.ts
import { IsInt, Min, IsOptional, IsString, MaxLength } from 'class-validator';
import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';

export class CreateDepositDto {
  @ApiProperty({ description: '充值金额（整数，单位：元）', example: 500 })
  @IsInt()
  @Min(1)
  amount: number;

  @ApiPropertyOptional({ description: '备注', example: '' })
  @IsOptional()
  @IsString()
  @MaxLength(200)
  remark?: string;
}
```

**调用链：**

```typescript
@Post('create')
@UseGuards(JwtAuthGuard)
@ApiBearerAuth()
@HttpCode(HttpStatus.OK)
@ApiOperation({ summary: '创建押金充值订单' })
async createDeposit(
  @Req() req: AuthenticatedRequest,
  @Body() dto: CreateDepositDto,
) {
  const userId = req.user!.id;
  const result = await this.depositService.createDeposit(userId, dto.amount);
  return { success: true, data: result };
}
```

**成功响应：**

```json
{
  "success": true,
  "data": {
    "depositId": "uuid-string",
    "orderNo": "DEPXXXXXXXXXXXX",
    "paymentNo": "PAYXXXXXXXXXXXX",
    "amount": 500
  }
}
```

**错误响应：**
- 400：金额必须大于0 / 只能充值整数金额
- 401：未授权

### 2.5 完整控制器代码参考

以下是 `deposit.controller.ts` 的完整代码结构（供 Claude Code 直接编写）：

```typescript
import {
  Controller,
  Post,
  Get,
  Body,
  Req,
  UseGuards,
  HttpCode,
  HttpStatus,
  UnauthorizedException,
} from '@nestjs/common';
import {
  ApiTags,
  ApiOperation,
  ApiResponse,
  ApiBearerAuth,
} from '@nestjs/swagger';
import type { Request } from 'express';
import { DepositService } from './deposit.service';
import { CreateDepositDto } from './dto/create-deposit.dto';
import { ListDepositDto } from './dto/list-deposit.dto';
import { RefundDepositDto } from './dto/refund-deposit.dto';
import { JwtAuthGuard } from '../auth/jwt-auth.guard';

interface AuthenticatedRequest extends Request {
  user?: { id: string };
}

@ApiTags('押金')
@Controller('api/v1/deposit')
export class DepositController {
  constructor(private readonly depositService: DepositService) {}

  @Post('create')
  @UseGuards(JwtAuthGuard)
  @ApiBearerAuth()
  @HttpCode(HttpStatus.OK)
  @ApiOperation({ summary: '创建押金充值订单' })
  @ApiResponse({ status: 200, description: '创建成功' })
  @ApiResponse({ status: 400, description: '参数错误' })
  @ApiResponse({ status: 401, description: '未授权' })
  async createDeposit(
    @Req() req: AuthenticatedRequest,
    @Body() dto: CreateDepositDto,
  ) {
    const userId = req.user?.id;
    if (!userId) throw new UnauthorizedException();
    const result = await this.depositService.createDeposit(userId, dto.amount);
    return { success: true, data: result };
  }

  @Post('list')
  @UseGuards(JwtAuthGuard)
  @ApiBearerAuth()
  @HttpCode(HttpStatus.OK)
  @ApiOperation({ summary: '押金记录列表' })
  @ApiResponse({ status: 200, description: '查询成功' })
  @ApiResponse({ status: 401, description: '未授权' })
  async listDeposits(
    @Req() req: AuthenticatedRequest,
    @Body() dto: ListDepositDto,
  ) {
    const userId = req.user?.id;
    if (!userId) throw new UnauthorizedException();
    const page = dto.page ?? 1;
    const limit = dto.limit ?? 20;
    const result = await this.depositService.listDeposits(userId, page, limit);
    return { success: true, data: result };
  }

  @Get('balance')
  @UseGuards(JwtAuthGuard)
  @ApiBearerAuth()
  @HttpCode(HttpStatus.OK)
  @ApiOperation({ summary: '押金余额查询' })
  @ApiResponse({ status: 200, description: '查询成功' })
  @ApiResponse({ status: 401, description: '未授权' })
  @ApiResponse({ status: 404, description: '用户不存在' })
  async getBalance(@Req() req: AuthenticatedRequest) {
    const userId = req.user?.id;
    if (!userId) throw new UnauthorizedException();
    const result = await this.depositService.getBalance(userId);
    return { success: true, data: result };
  }

  @Post('refund')
  @UseGuards(JwtAuthGuard)
  @ApiBearerAuth()
  @HttpCode(HttpStatus.OK)
  @ApiOperation({ summary: '押金退款申请' })
  @ApiResponse({ status: 200, description: '申请成功' })
  @ApiResponse({ status: 400, description: '参数错误/押金不足/有未处理申请' })
  @ApiResponse({ status: 401, description: '未授权' })
  @ApiResponse({ status: 404, description: '用户不存在' })
  async refundDeposit(
    @Req() req: AuthenticatedRequest,
    @Body() dto: RefundDepositDto,
  ) {
    const userId = req.user?.id;
    if (!userId) throw new UnauthorizedException();
    const result = await this.depositService.refundDeposit(
      userId,
      dto.amount,
      dto.alipayNo,
      dto.alipayRealname,
    );
    return { success: true, data: result };
  }
}
```

## 4. 模块设计

### src/deposit/deposit.module.ts

```typescript
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { Deposit } from './deposit.entity';
import { DepositService } from './deposit.service';
import { DepositController } from './deposit.controller';
import { User } from '../users/user.entity';
import { Payment } from '../payments/payment.entity';

@Module({
  imports: [
    TypeOrmModule.forFeature([Deposit, User, Payment]),
  ],
  providers: [DepositService],
  controllers: [DepositController],
  exports: [DepositService],
})
export class DepositModule {}
```

**说明：**
- `TypeOrmModule.forFeature` 注册了 3 个实体：`Deposit`、`User`、`Payment`，因为 `DepositService` 需要注入这 3 个 Repository。
- `exports: [DepositService]` 允许其他模块（如竞标模块、支付模块）通过依赖注入调用 `handleDepositPaymentSuccess` 和 `deductDeposit` 等内部方法。

#### 5.2 在 @Module.imports 数组中添加

```typescript
@Module({
  imports: [
    // ... 现有模块 ...
    WarehouseModule,
    DepositModule,   // ← 新增这一行
  ],
  // ...
})
export class AppModule {}
```

#### 5.3 在 TypeORM entities 数组中添加 Deposit 实体

在 `TypeOrmModule.forRootAsync` 的 `entities` 数组中添加：

```typescript
entities: [
  User,
  Address,
  Product,
  Category,
  PriceHistory,
  Order,
  OrderItem,
  Cart,
  CartItem,
  Payment,
  ShipmentOrder,
  OrderPhoto,
  Deposit,    // ← 新增这一行
],
```

> **注意：** `Deposit` 实体目前没有被任何 TypeORM 配置注册。虽然在模块层 `TypeOrmModule.forFeature` 可以用于注入 Repository，但根模块的 `entities` 数组也必须包含它，否则迁移工具和 TypeORM CLI 无法识别该实体。

### 6.3 不对外暴露的 Service 方法

以下两个方法**不**在 Controller 中暴露，仅供其他模块内部调用：

| 方法 | 使用方 | 说明 |
|------|--------|------|
| `handleDepositPaymentSuccess(orderNo, paymentId)` | 支付模块回调处理 | 支付成功时调用 |
| `deductDeposit(userId, amount, remark)` | 竞标/订单模块 | 扣减用户押金 |

如果要让其他模块调用，需要将 `DepositService` 注入到目标模块中（已通过 `exports: [DepositService]` 暴露）。

### 6.4 路由汇总

| 方法 | 路由 | 鉴权 | 说明 |
|------|------|------|------|
| POST | `/api/v1/deposit/create` | JwtAuthGuard | 创建押金充值订单 |
| POST | `/api/v1/deposit/list` | JwtAuthGuard | 押金记录列表 |
| GET | `/api/v1/deposit/balance` | JwtAuthGuard | 押金余额查询 |
| POST | `/api/v1/deposit/refund` | JwtAuthGuard | 押金退款申请 |
